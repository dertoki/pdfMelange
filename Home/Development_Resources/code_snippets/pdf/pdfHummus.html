Some modifications are necessary to create a build of PDFWriter that depends on system libraries of libtiff, libjpeg and libFreeType.

Apply the patch with:

Paste +patch to PDFWriter.patch.


cd PDFWriterCMake3_2
patch -p1 -b < ../PDFWriter.patch
Do the build:
cd PDFWriterCMake3_2
mkdir Build
cd Build
cmake -DCMAKE_BUILD_TYPE=MinSizeRel ..
Test szenario:
#include "PDFWriter.h"

/**
  * Test the merging of pages.
  */
int main(void){

    PDFWriter pdfWriter;
    PDFPageRange selectivePageRange;
    selectivePageRange.mType = PDFPageRange::eRangeTypeSpecific;

    pdfWriter.StartPDF("hummus-test-merge.pdf",ePDFVersion13);

    selectivePageRange.mSpecificRanges.push_back(ULongAndULong(0,1));
    selectivePageRange.mSpecificRanges.push_back(ULongAndULong(8,9));
    pdfWriter.AppendPDFPagesFromPDF("123.pdf",selectivePageRange);

    selectivePageRange.mSpecificRanges.clear();
    selectivePageRange.mSpecificRanges.push_back(ULongAndULong(0,9));
    pdfWriter.AppendPDFPagesFromPDF("abc.pdf",selectivePageRange);

    pdfWriter.EndPDF();
    return 1;
}
