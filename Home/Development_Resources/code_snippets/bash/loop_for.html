loop for

Created Sonntag 07 Dezember 2014

An example with find:
for file in $(find $HOME -maxdepth 1 -name '.*'); 
do 
	readlink -f $file
done
Variables listing:
VAR_A="Otto"
VAR_B="Wilma"
VAR_C="Marwin"
for var in ${!VAR_*}
do 
	echo "Name: "${!var}
done
Array listing:
Pattern[1]='Max'
Pattern[2]='Michael'
Pattern[3]='Maria'
Pattern[4]='Marwin'

for iPattern in ${Pattern[@]} ; do
     echo "Name: "$iPattern
done
