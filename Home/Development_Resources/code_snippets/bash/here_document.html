...
EOF
"here document" with gcc:
DATUM=$(date)
gcc -Wall -x c - <<EOF
#include <stdio.h>
#include <string.h>
int main(void){
   int i;
   char line[] = "this executable was built at $DATUM by $USER";
   for (i=0; i<strlen(line); i++) putc('-', stdout);
   puts("\nHello World,");
   puts(line);
   for (i=0; i<strlen(line); i++) putc('-', stdout);
   putc('\n', stdout);
   return 1;
}
EOF
./a.out
rm a.out
