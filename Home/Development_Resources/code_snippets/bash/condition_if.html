M_PREFIX='/opt/gtk36'
M_HOST='i686-w64-mingw32'
M_BUILD='x86_64-linux-gnu'

M_CFLAGS="-I$M_PREFIX/include -Os -s"
M_CPPFLAGS="-I$M_PREFIX/include -Os -s"
M_CXXFLAGS="-I$M_PREFIX/include -Os -s"

# Do some system checks
#######################

# Check if directory M_PREFIX has write permissions.
if [ ! -d $M_PREFIX ] || [ ! -w $M_PREFIX ] ; then 
	echo "ERROR: Your directory M_PREFIX: \""$M_PREFIX"\" was NOT created with write permissions!"
	exit
fi

# Check if your machine hardware name (e.g. "x86_32") is a substring of your M_BUILD variable.
if [ $M_BUILD == ${M_BUILD/$(uname --machine)/} ] ; then
	echo "ERROR: Variable M_BUILD (building machine) \""$M_BUILD"\" does not fit to your installation!"
	echo "     : Shell command \"uname\" returns: \""$(uname --machine)"\"."
	exit
fi

# Check if the installed MinGW toolchain fits with your M_HOST (target host machine) variable.
if [ ! $(which $M_HOST-gcc) ] ; then
	echo "ERROR: Variable M_HOST (target host machine) \""$M_HOST"\" does not fit to your installation!"
	echo "     : Can't find your compiler: \""$M_HOST-gcc"\"."
	echo "     : Modify variable M_HOST or install the correct MinGW toolchain."
	exit
fi
